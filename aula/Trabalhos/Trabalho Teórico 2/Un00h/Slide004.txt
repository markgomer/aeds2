int fat (int n){
  int resp;
  if (n == 1){  /* Condição de parada */
    resp = 1;
  } else {
    resp = n * fat (n - 1); /* Chamada recursiva */
  }
  return resp;
}


int fib (int n){
  int resp;
  if (n == 0 || n == 1){  /* condição de parada */
    resp = 1;
  } else {
    resp = fib (n - 1) + fib(n - 2); /* chamada recursiva */
  }
  return resp;
}
