// Por que o código abaixo imprime 2, 1, 0, 0, 1 e 2?

void printRecursivo(){
  printRecursivo(2);
}

void printRecursivo(int i){
  System.out.println(i);
  if (i > 0){
    printRecursivo(i - 1);
  }
  System.out.println(i);
}


// todas as chamadas recursivas, após suas execuções, retornam à execução do programa principal com os valores anteriores de i .
